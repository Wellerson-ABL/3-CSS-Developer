Margin (margem)

		< style>
			div {
				width: 300px;
				height: 300px;
			}

			#elemento-01 {
				background: orange;

				margin-bottom: 20px;  	/* margin-bottom, define a margem inferior do elemento. ou seja, 
							o espaçamento entre o elemento e o proximo elemento */

				margin-left: 40px;  	/* margin-left, define a margem esquerda do elemento. ou seja, o 
							espaçamento entre o elemento e o proximo elemento */

				margin: 20px;  	/* margin, define a margem de todos os lados do elemento. ou seja, 
							o espaçamento entre o elemento e o proximo elemento */

				margin: 20px 20px;  	/* o primeiro valor define a margem superior e inferior, o segundo 
							valor para os lados */		

				margin: 20px 20px 20px; 	/* o primeiro valor define a margem superior, o segundo valor 
							define os lados, Terceiro valor define lado Inferior */

				margin: 20px 20px 20px 20px;	 /* o primeiro valor define a margem superior, o segundo 
							lado direito, Terceiro inferior, quarto lado esqueri(Segue sentido anti-horario) */

				margin: auto; 	/* auto, define a margem automaticamente. ou seja, o elemento vai ocupar 
							todo o espaço disponivel  (Centralizando)*/

				margin: inherit; 	/* inherit, herda o tamanho do elemento pai */

				margin: 20px auto; 	/* o primeiro valor define a margem superior e inferior, o segundo 
							valor para os lados (Centralizadona Horizontal)*/	
			}

			#elemento-02 {
				background: purple;

				margin: px; 	/* margin, define a margem de todos os lados do elemento. ou seja, o 
						espaçamento entre o elemento e o proximo elemento */

				margin-right: 50px; 	/* margin-right, define a margem direita do elemento. ou seja, 
							o espaçamento entre o elemento e o proximo elemento */

				margin-top: 20px; 	/* margin-top, define a margem superior do elemento. ou seja, 
							o espaçamento entre o elemento e o proximo elemento */

				margin-left: 50px; 	/* margin-left, define a margem esquerda do elemento. ou seja, 
							o espaçamento entre o elemento e o proximo elemento */
			}

		< /style>

			< div id="elemento-02">< /div>
