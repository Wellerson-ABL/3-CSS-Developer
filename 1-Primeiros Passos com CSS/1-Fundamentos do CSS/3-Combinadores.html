	< style>
		div + p {  /* Vao aplicar somente irmao adjacente, ou seja o proximo 'p = paragrafo 4' que estão fora da div. */
		background: blueviolet;
		color: white;	
	}
	div ~ p {  /* Vao aplicar em todos irmãos adjacentes, ou seja todos os proximos 'p = paragrafos' que estão fora da div. */
		background: blue;
		color: white;	
	}
	.fundo-verde span p { /* Vai aplicar somente 'p = paragrafo 3' está dentro da span. */
		background:green;
		color: yellow;
	} 
	.fundo-verde span ~ p { /* Vai aplicar somente irmao adjacente 'p = paragrafo 3.2', o segundo 'p' que está fora do span. */
		background:gray;
	} 
	.fundo-verde span + p { /* Vai aplicar somente irmao adjacente 'p = paragrafo 3.1' o primeiro 'p' que está fora da span. */
		background:pink;
	} 
	< /style>

	< div class="fundo-verde">
		< p>Paragrafo 1 Dentro da Div< /p>
		< p>Paragrafo-2 dentro da div< /p>
